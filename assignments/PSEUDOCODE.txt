GLOBAL VARIABLES
----------------
GREEN, RED, YELLOW, CYAN, BLUE, MAGENTA, RESET = Color codes for terminal
HIGHSCORE_FILE = Path to "highscore.txt" file


FUNCTION strip_ansi(text)
    INPUT: text (string that may contain color codes)
    OUTPUT: clean text without color codes
    
    USE regular expression to find all ANSI color codes
    REMOVE all found color codes from text
    RETURN cleaned text
END FUNCTION


FUNCTION box_line(content, width = 39)
    INPUT: content (text to display), width (box width)
    OUTPUT: formatted line with borders and centered text
    
    clean_content = strip_ansi(content)
    total_padding = width - length of clean_content
    left_pad = total_padding divided by 2 (round down)
    right_pad = total_padding - left_pad
    
    RETURN "║" + (left_pad spaces) + content + (right_pad spaces) + "║"
END FUNCTION


FUNCTION clear_screen()
    Clear the terminal/console display
END FUNCTION


FUNCTION print_centered(text, color = default)
    OUTPUT: text centered on 80-character wide screen
    
    PRINT text centered using the given color
END FUNCTION


FUNCTION print_separator()
    OUTPUT: a decorative line of 80 dashes
    
    PRINT 80 dashes in CYAN color
END FUNCTION


FUNCTION load_highscores()
    INPUT: none
    OUTPUT: list of top 5 hig   h scores (or empty list if no file)
    
    TRY
        OPEN highscore file for reading
        READ all lines from file
        
        scores = empty list
        FOR each line in file
            IF line is not empty THEN
                SPLIT line by comma to get: name, score, difficulty
                CONVERT score to integer
                ADD (name, score, difficulty) to scores list
            END IF
        END FOR
        
        SORT scores by score in descending order (highest first)
        RETURN only the first 5 scores
        
    CATCH FileNotFoundError
        RETURN empty list
    END TRY
END FUNCTION


FUNCTION save_highscore(name, score, difficulty)
    INPUT: name (player name), score (points), difficulty (game level)
    OUTPUT: writes updated high scores to file
    
    scores = load_highscores()
    ADD (name, score, difficulty) to scores list
    
    SORT scores by score in descending order
    KEEP only the first 5 scores
    
    OPEN highscore file for writing
    FOR each score entry in scores
        WRITE entry as: name,score,difficulty
    END FOR
    CLOSE file
END FUNCTION

FUNCTION show_title_screen()
    INPUT: none
    OUTPUT: displays welcome screen
    
    clear_screen()
    PRINT blank line
    PRINT top border box
    PRINT box_line with "✦ NUMBER GUESSING GAME ✦"
    PRINT bottom border box
    PRINT blank line
    PRINT centered: "Guess the mystery number!"
    PRINT centered: "Test your luck and intuition!"
END FUNCTION

FUNCTION choose_difficulty()
    INPUT: user input from player
    OUTPUT: tuple (min_number, max_number, attempts, difficulty_name)
    
    LOOP forever
        print_separator()
        PRINT "SELECT DIFFICULTY"
        print_separator()
        PRINT difficulty options:
            1. Easy   (1–10, 5 attempts)
            2. Medium (1–50, 7 attempts)
            3. Hard   (1–100, 10 attempts)
        
        choice = GET user input
        
        IF choice is "1" THEN
            RETURN (1, 10, 5, "Easy")
        ELSE IF choice is "2" THEN
            RETURN (1, 50, 7, "Medium")
        ELSE IF choice is "3" THEN
            RETURN (1, 100, 10, "Hard")
        ELSE
            PRINT error message "Invalid choice. Try again."
        END IF
    END LOOP
END FUNCTION

FUNCTION play_game(player_name)
    INPUT: player_name (string)
    OUTPUT: tuple (final_score, difficulty)
    
    (low, high, attempts, difficulty) = choose_difficulty()
    
    secret = random number between low and high
    score = 100
    
    PRINT blank line
    PRINT centered: "🎮 {player_name}'s Turn!"
    print_separator()
    PRINT "I'm thinking of a number between {low} and {high}."
    PRINT "You have {attempts} attempts."
    
    FOR attempt = 1 to attempts DO
        guess = GET user input (current attempt number)
        
        IF guess is not a number THEN
            PRINT error: "Please enter a number."
            CONTINUE to next iteration
        END IF
        
        guess = convert guess to integer
        
        IF guess < low OR guess > high THEN
            PRINT error: "Please enter a number between {low} and {high}."
            CONTINUE to next iteration
        END IF
        
        IF guess equals secret THEN
            PRINT success: "✓ Correct! You guessed the number!"
            PRINT "Your score: {score}"
            RETURN (score, difficulty)
        ELSE IF guess < secret THEN
            PRINT hint: "↑ Too low!"
        ELSE
            PRINT hint: "↓ Too high!"
        END IF
        
        score = score - 10
        PRINT blank line
    END FOR
    
    PRINT "✗ You ran out of attempts! The number was {secret}."
    PRINT "Your score: 0"
    RETURN (0, difficulty)
END FUNCTION

FUNCTION main()
    INPUT: none
    OUTPUT: none (runs entire game)
    
    show_title_screen()
    
    highscores = load_highscores()
    IF highscores is not empty THEN
        print_separator()
        PRINT centered: "TOP 5 HIGH SCORES"
        print_separator()
        FOR rank = 1 to number of highscores DO
            (name, score, difficulty) = highscores[rank]
            PRINT "{rank}. {name} {score} points ({difficulty})"
        END FOR
        PRINT blank line
    END IF
    
    player_name = GET user input "Enter your name: "
    PRINT blank line
    
    LOOP forever
        (score, difficulty) = play_game(player_name)
        
        IF fewer than 5 high scores exist OR score > lowest high score THEN
            PRINT "🏆 New High Score! 🏆"
            save_highscore(player_name, score, difficulty)
            highscores = load_highscores()
        END IF
        
        again = GET user input "Play again? (y/n): "
        
        IF again is not "y" THEN
            print_separator()
            PRINT centered: "Thanks for playing! Goodbye!"
            print_separator()
            BREAK from loop (exit game)
        END IF
    END LOOP
END FUNCTION


IF this file is the main program THEN
    CALL main()
END IF
